26/**
27 * @class Coordinator
28 * @brief Coordinates entities, components, and systems.
29 *
30 * This version defers modifications (destroyEntity, addComponent, removeComponent)
31 * via a thread-safe command queue. Modifications are enqueued (async) and applied
32 * when flushCommands() is called (typically once per frame).
33 *
34 * For cases where you require immediate (synchronous) execution you have two options:
35 * - Call the alternative sync methods (e.g. addComponentSync).
36 * - Turn off async modifications globally by calling setAsyncModifications(false).
37 *
38 * Note on getComponent: This method returns a reference for direct read/edit access.
39 * If you plan to modify a component from multiple threads, you should either perform
40 * the modification only after flushCommands() or provide a dedicated update method
41 * that enqueues modifications.
42 */

208 /**
209 * @brief Checks if an entity has a specific component
210 * @tparam T Component type to check for
211 * @param entity Entity to check
212 * @param force If true, bypasses mutex lock for performance (use with caution)
213 * @return True if the entity has the component, false otherwise
214 */
215 template <typename T>
216 bool hasComponent(Entity entity, bool force = false) {
217 if (force) {
218 // Skip mutex lock if force=true (use with caution)
219 auto signature = entityManager->getSignature(entity);
220 auto compTypeID = componentManager->getComponentTypeID<T>();
221 return signature.test(compTypeID);
222 } else {
223 std::lock_guard<std::shared_mutex> lock(m_ecsMutex);
224 auto signature = entityManager->getSignature(entity);
225 auto compTypeID = componentManager->getComponentTypeID<T>();
226 return signature.test(compTypeID);
227 }
228 }

230 /**
231 * @brief Checks if a pair of entities has a specific pair of components
232 * @tparam T1 First component type
233 * @tparam T2 Second component type
234 * @param entityA First entity
235 * @param entityB Second entity
236 * @return True if the entities have the components (in either order), false otherwise
237 */
238 template <typename T1, typename T2>
239 bool hasComponentPair(Entity entityA, Entity entityB) {
240 std::lock_guard<std::shared_mutex> lock(m_ecsMutex);
241 return (hasComponent<T1>(entityA, true) && hasComponent<T2>(entityB, true)) ||
242 (hasComponent<T2>(entityA, true) && hasComponent<T1>(entityB, true));
243 }

267 * @tparam T2 Second component type to check for

272 std::lock_guard<std::shared_mutex> lock(m_ecsMutex);
273 std::vector<Entity> entitiesWithComponents;
274 for (auto entity : entityManager->getEntities()) {
275 if (hasComponent<T1>(entity, true) && hasComponent<T2>(entity, true)) {
